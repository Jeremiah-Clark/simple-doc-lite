% Tries the primary font first; if it isn't installed, falls back to the

% last resort.  A console message is printed when a fallback is used.

  \usepackage{fontspec}

  % ── Body font ──

    \setmainfont{$font-body$}%
  }{%
    \typeout{Simple Doc: body font "$font-body$" not found — using fallback.}%
    \setmainfont{$if(font-body-fallback)$$font-body-fallback$$else$Noto Sans$endif$}%
  }
$else$
  \setmainfont{$if(font-body-fallback)$$font-body-fallback$$else$Noto Sans$endif$}
$endif$

  % ── Heading font ──
$if(font-heading)$
  \IfFontExistsTF{$font-heading$}{%

  }{%
    \typeout{Simple Doc: heading font "$font-heading$" not found — using fallback.}%
    \setsansfont{$if(font-heading-fallback)$$font-heading-fallback$$else$Noto Sans$endif$}%
  }
$else$
  \setsansfont{$if(font-heading-fallback)$$font-heading-fallback$$else$Noto Sans$endif$}
$endif$

  % ── Monospace font ──

    \setmonofont{$if(font-mono-fallback)$$font-mono-fallback$$else$Noto Sans Mono$endif$}[Scale=MatchLowercase]%

% ---------------------------------------------------------------------------
% Callout / admonition environments  (mdframed)
% ---------------------------------------------------------------------------
% Each environment accepts an optional argument to override the default title:
%   \begin{example}                    → title bar says "Example"
%   \begin{example}[Hand Count Reset]  → title bar says "Hand Count Reset"
%
% Helper macro: \definecallout{envname}{Default Title}{base-color}
% Derives border/title bar (color!80!black) and background (color!5!white)
% from a single base color.

\newcommand{\definecallout}[3]{%
  \newenvironment{#1}[1][#2]{%
    \begin{mdframed}[
      linecolor=#3!80!black, backgroundcolor=#3!5!white,
      linewidth=2pt, roundcorner=2mm,
      innertopmargin=10pt, innerbottommargin=10pt,
      innerleftmargin=10pt, innerrightmargin=10pt,
      skipabove=9pt, skipbelow=6pt, nobreak=true,
      frametitle=##1, frametitlebackgroundcolor=#3!80!black,
      frametitlefont=\bfseries\color{white},
      frametitleaboveskip=4pt, frametitlebelowskip=5pt
    ]%
  }{\end{mdframed}}%
}
