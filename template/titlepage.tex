% titlepage.tex — Two layouts, selected by short-form: in master.yaml.
%
% Available commands (from template.tex):
%   \doctitle, \docauthor, \docdate
%   \docversion             — only if version: is set
%   \doclogo                — only if logo: is set
%   \docdisclaimer          — only if disclaimer: is set (long-form only)
%   \doclogowidth           — long-form logo width (fraction of \textwidth)
%   \doctppostrule, \doctpposttitle, \doctppostauthor, \doctppostdate
%   \docshortform           — only if short-form: true
%   \docsfimageheight       — short-form banner max height

\ifdefined\docshortform
  % ══════════════════════════════════════════════════════════
  % SHORT-FORM: header block at top of page 1
  % ══════════════════════════════════════════════════════════
  \thispagestyle{firstpage}

  \ifdefined\doclogo
    \begin{center}
      \includegraphics[width=\linewidth,height=\docsfimageheight,keepaspectratio]{\doclogo}
    \end{center}
    \vspace{0.25em}
  \fi

  {\noindent\color{black!50}\rule{\linewidth}{0.4pt}}
  \vspace{0.35em}

{\noindent\LARGE\sffamily\bfseries\color{brandblue} \doctitle}

\vspace{0.4em}

\noindent
\begin{minipage}[t]{0.60\linewidth}
  \vspace{0pt}
  {\sffamily \docauthor}
\end{minipage}%
\hfill
\begin{minipage}[t]{0.38\linewidth}
  \vspace{0pt}
  \raggedleft
  {\sffamily \docdate}%
  \ifdefined\docversion
    \quad{\sffamily\color{black!60} v\docversion}%
  \fi
\end{minipage}

  \vspace{0.45em}
  {\noindent\color{black!50}\rule{\linewidth}{0.4pt}}
  \vspace{1.5em}

\else
  % ══════════════════════════════════════════════════════════
  % LONG-FORM: full title page on its own page (default)
  % ══════════════════════════════════════════════════════════
  \begin{titlepage}
    \thispagestyle{empty}

    \ifdefined\docversion
      \begin{flushright}
        \vspace*{-2.5em}
        {\small\sffamily\color{black!60} v\docversion}
      \end{flushright}
      \vspace{-2em}
    \fi

    \begin{center}
      \rule{\linewidth}{0.4pt}
      \vspace{\doctppostrule}

      {\Huge\sffamily \doctitle}

      \vspace{\doctpposttitle}
      {\large\sffamily \docauthor}

      \vspace{\doctppostauthor}
      {\docdate}

      \vspace{\doctppostdate}

      \ifdefined\doclogo
        \includegraphics[width=\doclogowidth\textwidth]{\doclogo}
      \fi

      \vfill

      \ifdefined\docdisclaimer
        \begin{important}
        \docdisclaimer
        \end{important}
      \fi
    \end{center}
  \end{titlepage}
\fi
